- Reducer

        import C from './constants';
        import { combineReducers } from 'redux';

        export const param1 = (state = [], action) => {
                switch (action.type) {
                        case C.UPDATE_PARAM1:
                                return action.payload;
                        default:
                                return state;
                }
        }

        export const param2 = (state = false, action) => {

                switch (action.type) {

                        case C.SET_PARAM2:
                        return true

                        case C.UNSET_PARAM2:
                        return false

                        default:
                        return state
                }

        }

        export default combineReducers({
                param1,
                param2
        })
